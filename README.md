# Signals

## 1. 20 kHz Sine wave modulated with FM
fm = 10 kHz
fc = 100 kHz
df = 75 kHz
B = 2 * (fm + df) = 170 kHz
f_max = fc + df = 175 kHz
f_min = fc + df = 25 kHz

## 2. 20 kHz Square wave modulated with FSK
fm = 10 kHz
fc = 275 kHz
df = 75 kHz
B = 2 * (fm + df) = 170 kHz
f_max = fc + df = 200 kHz
f_min = fc - df = 350 kHz

## 3. 20 kHz Sawtooth wave modulated with AM
fm = 10 kHz
fc = 385 kHz
B = 2 * fm = 20 kHz
f_max = fc + fm = 395 kHz
f_min = fc - fm = 375 kHz

## 4. 20 kHz Square wave OOK
fm = 10 kHz
fc = 440 kHz
B = 2 * fm = 20 kHz
f_max = fc + fm = 450 kHz
f_min = fc - fm = 430 kHz
